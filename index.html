function countSmileys(arr) {
    arr = arr.join('');
    console.log(arr)
    let n = arr.length;
    let result = 0;
    for(let i = 0; i < n-1; i++)
      {
        if(arr[i] == ":" || arr[i] == ";")
          {
            if(arr[i+1] == "-" || arr[i+1] == "~")
              if(arr[i+2] == ")" || arr[i+2] == "D")
                result++;
            if(arr[i+1] == ")" || arr[i+1] == "D")
              result++;
          }
      }
    return result;
}
